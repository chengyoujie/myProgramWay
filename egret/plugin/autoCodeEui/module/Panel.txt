    constructor(){

    }

       self._shape = UIManager.createShape(0,0.5);
    }

    public showPanel():void {

        UIManager.ins.popToParnent(self._shape, App.ins.layer.commonLayer);
        UIManager.ins.popToParnent(self, App.ins.layer.commonLayer);

    }

        UIManager.ins.removeFromParnent(this._shape, App.ins.layer.commonLayer);
        UIManager.ins.removeFromParnent(self, App.ins.layer.commonLayer);
        self._shape.offclk(self.hidePanel,self);
    }
}
