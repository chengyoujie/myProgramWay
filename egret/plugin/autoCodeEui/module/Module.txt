class ${shortName}Module extends ModuleBase {

    private _mediator:${shortName}Mediator;
    private _service:${shortName}Service;
    
    constructor() {
        super("${shortName}Module");
    }

    public startup():void
    {
        let self = this;
        self._mediator = new ${shortName}Mediator(self);
        self._service = new ${shortName}Service(self);
        self.addEventListeners();
    }

    public shutDown():void
    {
        let self = this;
        self.removeEventListeners();
    }


    protected addEventListeners()
    {
        let self = this;
        cqobs.on(ModuleEventType.CLICK_MODULEID, self.handleOpen, self);
    }
    protected removeEventListeners():void
    {
        let self = this;
        cqobs.off(ModuleEventType.CLICK_MODULEID, self.handleOpen, self);
    }

    private handleOpen(e:GameEvent)
    {
        let data = e.data;
        if(data == ModuleID.${moduleID})
        {
            this._mediator.startup();
            this._mediator.showPanel();
        }
    }
}
